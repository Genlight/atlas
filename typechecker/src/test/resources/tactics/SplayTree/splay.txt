        (w (let:tree:cf leaf node))
        Q2zig@(ite (* a == b *)
          (w (let:tree:cf node node))
          (ite (* a < b *)
            (match (* bl, zig zig *)
              (w (let:tree:cf leaf (let:tree:cf node node)))
              (w

                  Q3zigzig@(let:tree:cf

                  )
                )
              )
            )
            (match (* br, zig zag *)

            )
          )
        )
      )
      (match (* cr, zag *)
        (w (let:tree:cf leaf node))
        Q2zag@(ite (* a == b *)

          (ite (* a < b *)
            (match (* bl, zag zig *)
              (w (let:tree:cf leaf (let:tree:cf node node)))
